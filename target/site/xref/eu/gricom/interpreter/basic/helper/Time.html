72      /**
73       * Parse an entered date + time into the object format.
74       * 
75       * @param strDate
76       *            - Date + time as string
77       * @throws Exception
78       *             - parsing fails
79       */
80      public final void parseDateTime(final String strDate) throws Exception {
81          // assumption that all dates in ProvJs are formatted according to the
82          // same pattern: YYYY-MM-DD HH:mm:ss, where HH is the hour from 00 - 23.
83          // If this pattern is broken, the method will return a 'false'.
84          parseDate(strDate);
85  
86          String strDummy;
87  
88          // first - isolate the hour
89          strDummy = strDate.substring(11, 13);
90          if (strDummy != null) {
91              _iHour = Integer.parseInt(strDummy);
92          } else {
93              throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Hour could not been parsed");
94          }
95  
96          // second - parse the month
97          strDummy = strDate.substring(14, 16);
98          if (strDummy != null) {
99              _iMinute = Integer.parseInt(strDummy);
100         } else {
101             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Minute could not been parsed");
102         }
103 
104         // third - parse the day
105         strDummy = strDate.substring(17, 19);
106         if (strDummy != null) {
107             _iSecs = Integer.parseInt(strDummy);
108         } else {
109             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Seconds could not been parsed");
110         }
111 
112         if ((_iHour < 0) || (_iHour > 23) || (_iMinute < 0) || (_iMinute > 59) || (_iSecs < 0) || (_iSecs > 59)) {
113             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Time out of range");
114         }
115 
116         _oDate.set(_iYear, _iMonth, _iDay, _iHour, _iMinute, _iSecs);
117     }

121      * 
122      * @param strDate
123      *            - Date as string
124      * @throws Exception
125      *             - parsing fails
126      */
127     public final void parseDate(final String strDate) throws Exception {
128         // assumption that all dates in ProvJs are formatted according to the
129         // same pattern: YYYY-MM-DD.
130         // If this pattern is broken, the method will return a 'false'.
131         String strDummy;
132 
133         // first - isolate the year
134         strDummy = strDate.substring(0, 4);
135         if (strDummy != null) {
136             _iYear = Integer.parseInt(strDummy);
137         } else {
138             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Year could not been parsed");
139         }
140 
141         // second - parse the month
142         strDummy = strDate.substring(5, 7);
143         if (strDummy != null) {
144             _iMonth = Integer.parseInt(strDummy);
145         } else {
146             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Month could not been parsed");
147         }
148 
149         // third - parse the day
150         strDummy = strDate.substring(8, 10);
151         if (strDummy != null) {
152             _iDay = Integer.parseInt(strDummy);
153         } else {
154             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Day could not been parsed");
155         }
156 
157         if ((_iYear < 0) || (_iYear > 3000) || (_iMonth < 1) || (_iMonth > 12) || (_iDay < 0) || (_iDay > 31)) {
158             throw new Exception("[eu.gricom.interpreter.basic.helper.Time] Date out of range");
159         }
160 
161         _oDate.set(_iYear, _iMonth, _iDay);
162     }

312     /**
313      * Adds a number of days to this time object.
314      * 
315      * @param iDays
316      *            - number of days to add
317      */
318     public final void addDays(final int iDays) {
319 
320         int iMaxDays = getDaysInMonth(_iMonth, _iYear);
321 
322         _iDay += iDays;
323         // calculate a balance of months and days from the input parameter
324         while (_iDay > iMaxDays) {
325             _iDay -= iMaxDays;
326             _iMonth++;
327             if (_iMonth > 12) {
328                 _iMonth -= 12;
329                 _iYear++;
330             }
331             iMaxDays = getDaysInMonth(_iMonth, _iYear);
332         }
333 
334         _oDate.set(_iYear, _iMonth - 1, _iDay, _iHour, _iMinute, _iSecs);
335     }

337     /**
338      * Adds a number of months to this time object.
339      * 
340      * @param iInputMonths
341      *            - number of months to add
342      */
343     public final void addMonths(final int iInputMonths) {
344         int iMonth = _iMonth + iInputMonths;
345 
346         while (iMonth > 12) {
347             _iYear++;
348             iMonth--;
349         }
350 
351         // now normailze the day again ...
352         int iMaxDays = getDaysInMonth(iMonth, _iYear);
353         while (_iDay > iMaxDays) {
354             iMonth++;
355             _iDay -= iMaxDays;
356             iMaxDays = getDaysInMonth(iMonth, _iYear);
357         }
358 
359         _oDate.set(_iYear, iMonth - 1, _iDay, _iHour, _iMinute, _iSecs);
360         _iMonth = iMonth;
361     }

460     /**
461      * Difference between to dates in days
462      * 
463      * This method uses the date represented by this time object and compares it
464      * with another time object.
465      * 
466      * @param oTime
467      *            - the time object to compare to
468      * @return - difference in days
469      */
470     public final int diffInDays(final Time oTime) {
471         GregorianCalendar oGregorianCalendar = new GregorianCalendar();
472         int iDifference;
473         Time oLowerTime;
474         Time oHigherTime;
475 
476         if (matches(oTime)) {
477             return (0);
478         }
479 
480         if (greaterThan(oTime)) {
481             oLowerTime = oTime;
482             oHigherTime = this;
483         } else {
484             oLowerTime = this;
485             oHigherTime = oTime;
486         }
487 
488         iDifference = getDaysUntilEndOfYear(oLowerTime);
489 
490         int iWorkYear = oLowerTime._iYear + 1;
491         while (iWorkYear < (oHigherTime._iYear - 1)) {
492             if (oGregorianCalendar.isLeapYear(iWorkYear)) {
493                 iDifference += 366;
494             } else {
495                 iDifference += 365;
496             }
497         }
498 
499         return (iDifference);
500     }

502     /**
503      * Get the number of days in a month.
504      * 
505      * @param iMonth
506      *            - the month requested (1 - 12)
507      * @param iYear
508      *            - the year requested
509      * @return - number of days in month
510      */
511     private int getDaysInMonth(final int iMonth, final int iYear) {
512         GregorianCalendar oGregorianCalendar = new GregorianCalendar();
513 
514         int iMaxDays = 31;
515 
516         // check whether the month is February ...
517         if (iMonth == 2) {
518             // now that is February ... let's see how many days February has ...
519             if (oGregorianCalendar.isLeapYear(iYear)) {
520                 iMaxDays = 29;
521             } else {
522                 iMaxDays = 28;
523             }
524         }
525 
526         if ((iMonth == 4) || (iMonth == 6) || (iMonth == 9) || (iMonth == 11)) {
527             iMaxDays = 30;
528         }
529         return (iMaxDays);
530     }
