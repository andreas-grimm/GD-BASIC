32   * Description:
33   * <p>

39  @SuppressWarnings("SpellCheckingInspection")
40  public class Basic {
41      private static BufferedReader _oLineIn;
42      private final Logger _oLogger = new Logger(this.getClass().getName());
43      private static String _strBasicVersion = "basic";
44      private LineNumberStatement _aoLineNumbers = new LineNumberStatement();
45  
46      /**
47       * Constructs a new Basic instance. The instance stores the global state of
48       * the interpreter such as the values of all of the variables and the
49       * current statement.
50       */
51      public Basic() {
52          InputStreamReader oConverter = new InputStreamReader(System.in);
53          _oLineIn = new BufferedReader(oConverter);
54      }
55  
56      /**
57       * Interpret.
58       * This is where the magic happens. This runs the code through the parsing pipeline to generate the AST. Then it
59       * executes each statement. It keeps track of the current line in a member variable that the statement objects
60       * have access to. This lets "goto" and "if then" do flow control by simply setting the index of the current statement.
61       *
62       * In an interpreter that didn't mix the interpretation logic in with the AST node classes, this would be doing a lot more work.
63       *
64       * TODO: Convert to stand-alone program
65       *
66       * @param strProgram A string containing the source code of a .bas script to interpret.
67       */
68      public final void interpret(final String strProgram) {
69          ProgramPointer oProgramPointer = new ProgramPointer();
70          List<Statement> aoStatements = null;
71  
72          // Tokenize. At the end of the tokenization I have the program transferred into a list of tokens and parameters
73          _oLogger.info("Starting tokenization...");
74  
75          Tokenizer oTokenizer;
76  
77          if (_strBasicVersion.contains("jasic")) {
78              oTokenizer = new JasicLexer();
79          } else {
80              oTokenizer = new BasicLexer();
81          }
82  
83          List<Token> aoTokens = null;
84  
85          try {
86              aoTokens = oTokenizer.tokenize(strProgram);
87  
88          } catch (SyntaxErrorException e) {
89              System.out.println(e.getMessage());
90              System.exit(0);
91          }
92  
93          int iCounter = 0;
94          for (Token oToken: aoTokens) {
95              if (oToken.getType().toString().contains("LINE")) {
96                  _oLogger.debug("[" + oToken.getLine() + "] Token # <" + iCounter + ">: [" + oToken.getType().toString() + "]: []");
97              } else {
98                  _oLogger.debug("[" + oToken.getLine() + "] Token # <" + iCounter + ">: [" + oToken.getType().toString() + "]: [" + oToken.getText() + "]");
99              }
100             iCounter++;
101         }
102 
103         // Parse.
104         _oLogger.info("Starting parsing...");
105         try {
106             Parser oParser;
107 
108             if (_strBasicVersion.contains("jasic")) {
109                 oParser = new JasicParser(aoTokens);
110             } else {
111                 oParser = new BasicParser(aoTokens);
112             }
113 
114             aoStatements = oParser.parse();
115         } catch (SyntaxErrorException eSyntaxError) {
116             eSyntaxError.printStackTrace();
117         }
118 
119         // Run.
120         _oLogger.info("Starting execution...");
121         try {
122             if (aoStatements != null) {
123                 while (oProgramPointer.getCurrentStatement() < aoStatements.size()) {
124                     // as long as we have not reached the end of the code
125                     int iThisStatement = oProgramPointer.getCurrentStatement();
126                     int iSourceCodeLineNumber = _aoLineNumbers.getLineNumber(aoStatements.get(iThisStatement).getLineNumber());
127                     oProgramPointer.calcNextStatement();
128                     _oLogger.debug("Basic Source Code Line ["+ iSourceCodeLineNumber + "] Statement [ " + aoStatements.get(iThisStatement).getLineNumber() +"]: " + aoStatements.get(iThisStatement).content());
129                     aoStatements.get(iThisStatement).execute();
130                     iThisStatement = oProgramPointer.getCurrentStatement();
131                     _oLogger.debug("Next Source Code Line ["+ iSourceCodeLineNumber + "] Statement [ " + aoStatements.get(iThisStatement).getLineNumber() +"]: " + aoStatements.get(iThisStatement).content());
132                 }
133             } else {
134                 _oLogger.error("Parsing delivered empty program");
135             }
136         } catch (Exception eException) {
137             eException.printStackTrace();
138         }
139     }

144     /**
145      * Runs the interpreter as a command-line app. Takes one argument: a path
146      * to a script file to load and run. The script should contain one
147      * statement per line.
148      *
149      * @param args Command-line arguments.
150      */
151     public static void main(final String[] args) {
152         Logger oLogger = new Logger("main");
153         oLogger.setLogLevel("");
154 
155         boolean bParseOK = true;
156         String strParseError = null;
157         CommandLine oCommandLine = null;
158 
159         // create Options object
160         Options options = new Options();
161 
162         try {
163             options.addOption("h", false, "help (This screen)");
164             options.addOption("i", true, "define input file");
165             options.addOption("q", false, "quiet mode");
166             options.addOption("v", true, "verbose level: (0 - 3");
167             options.addOption("b", true, "BASIC type (J = Jasic, B = Basic");
168 
169             CommandLineParser parser = new DefaultParser();
170             oCommandLine = parser.parse(options, args);
171         } catch (ParseException exParseException) {
172             bParseOK = false;
173             // TODO: This here makes no sense...
174             strParseError = exParseException.getMessage();
175         }
176 
177         if ((oCommandLine != null) && (!oCommandLine.hasOption("q"))) {
178 
179             Printer.println("GriCom Basic Interpreter Version 0.1.0");
180             Printer.println("(c) Copyright Bob Nystrom 2010");
181             Printer.println("(c) Copyright Andreas Grimm 2020");
182 
183             long lMaxMemory = Runtime.getRuntime().maxMemory();
184             Printer.println("Maximum memory (bytes): " + lMaxMemory + ", ");
185             Printer.println("Free memory (bytes): " + Runtime.getRuntime().freeMemory());
186         }
187 
188         if ((oCommandLine != null) && (oCommandLine.hasOption("v"))) {
189             String strLogLevel = oCommandLine.getOptionValue("v");
190             String strLogLevelList = "trace|debug|info|warning";
191 
192             if (strLogLevelList.contains(strLogLevel.toLowerCase(Locale.ROOT))) {
193                 oLogger.setLogLevel(strLogLevel);
194             }
195 
196             oLogger.debug("Log Level set:" + strLogLevel + "...");
197         }
198 
199         if ((oCommandLine != null) && (oCommandLine.hasOption("h"))) {
200             // automatically generate the help statement
201             oLogger.debug("Display help message...");
202 
203             HelpFormatter formatter = new HelpFormatter();
204             formatter.printHelp("java -jar BASIC-<build-name>.jar -i <filename.bas>", options);
205         }
206 
207         if (!bParseOK) {
208             oLogger.error("Unknown command line parameter...");
209             System.exit(-1);
210         }
211 
212         // Just show the usage and quit if a script wasn't provided.
213         if ((oCommandLine != null) && (!oCommandLine.hasOption("i"))) {
214             oLogger.error("Program file name missing...");
215             Printer.println("");
216             Printer.println("usage: java -jar BASIC-<build-name>.jar -i <filename.bas>");
217             Printer.println("where <filename.bas> is a relative path to a .bas program to run.");
218             System.exit(-1);
219         }
220 
221         if ((oCommandLine != null) && (oCommandLine.hasOption("b"))) {
222             // get the Basic Version
223             oLogger.debug("Get BASIC version...");
224             String strBasicVersion = oCommandLine.getOptionValue("b").toLowerCase(Locale.ROOT);
225             String strBasicVersionList = "jasic|basic";
226 
227             if (strBasicVersionList.contains(strBasicVersion.toLowerCase(Locale.ROOT))) {
228                 _strBasicVersion = strBasicVersion.toLowerCase(Locale.ROOT);
229             }
230 
231             oLogger.debug("Basic version set: " + _strBasicVersion + "...");
232         }
233 
234         String strProgram = oCommandLine.getOptionValue("i");
235 
236         // Read the file.
237         oLogger.info("Read file: " + strProgram + "...");
238         String strContents = FileHandler.readFile(strProgram);
239 
240         // Run it.
241         oLogger.info("Run the interpreter...");
242         Basic oBasic = new Basic();
243         oBasic.interpret(strContents);
244     }
