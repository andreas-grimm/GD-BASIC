25  /**
26   * This defines the Basic parser. The parser takes in a sequence of tokens
27   * and generates an abstract syntax tree. This is the nested data structure
28   * that represents the series of statements, and the expressions (which can
29   * nest arbitrarily deeply) that they evaluate. In technical terms, what we
30   * have is a recursive descent parser, the simplest kind to hand-write.
31   *
32   * As a side-effect, this phase also stores off the line numbers for each
33   * label in the program. It's a bit gross, but it works.
34   */
35  public class BasicParser implements Parser {
36      private final Logger _oLogger = new Logger(this.getClass().getName());
37      private final List<Token> _aoTokens;
38      private int _iPosition;
39      private LineNumberStatement oLineNumber = new LineNumberStatement();
40  
41      /**
42       * Default constructor.
43       * The constructor receives the tokenized program and parses it.
44       *
45       * @param aoTokens - the tokenized program
46       */
47      public BasicParser(final List<Token> aoTokens) {
48          _aoTokens = aoTokens;
49          _iPosition = 0;
50      }
51  
52      @Override
53      public List<Statement> parse() throws SyntaxErrorException {
54          LabelStatement oLabelStatement = new LabelStatement();
55          List<Statement> aoStatements = new ArrayList<>();
56  
57          int iOrgPosition;
58  
59          _oLogger.debug("Start parsing...");
60          boolean _bContinue = true;
61  
62          while (_bContinue) {
63              switch (getToken(0).getType()) {
64                  // COMMENT Token: Ignore any following part of the line, identical to the REM token.
65                  case COMMENT:
66                      _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [COMMENT] ");
67                      _iPosition++;
68                      break;
69  
70                  // END Token: Terminate execution of program
71                  case END:
72                      _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [END] ");
73                      oLineNumber.putLineNumber(getToken(0).getLine(), _iPosition);
74                      aoStatements.add(new EndStatement(_iPosition));
75                      _iPosition++;
76                      break;
77  
78                  // EOP Token: End of the program found, finishing the loop thru the program
79                  case EOP:
80                      _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [EOP] ");
81                      _bContinue = false;
82                      _iPosition++;
83                      break;
84  
85                  // GOTO Token: Read the line from terminal for processing
86                  case GOTO:
87                      _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [GOTO] ");
88                      iOrgPosition = _iPosition;
89                      oLineNumber.putLineNumber(getToken(0).getLine(), _iPosition);
90                      _iPosition++;
91                      String strLineNumber = consumeToken(TokenType.NUMBER).getText();
92                      aoStatements.add(new GotoStatement(iOrgPosition, strLineNumber));
93                      break;
94  
95                  case IF:
96                      iOrgPosition = _iPosition;
97                      oLineNumber.putLineNumber(getToken(0).getLine(), _iPosition);
98                      _iPosition++;
99                      Expression oCondition = expression();
100                     _oLogger.debug("-parse-> found Token: <" + (_iPosition -1) + "> [IF]: <" + oCondition.content() + ">");
101                     String strLabel = consumeToken(TokenType.THEN).getText();
102                     _oLogger.debug("-parse-> followed Token: <" + _iPosition + "> [THEN]: <" + strLabel + ">");
103                     Token oEndIfToken = findToken(TokenType.ENDIF);
104                     _oLogger.debug("-parse-> followed Token: <" + oEndIfToken.getLine() + "> [END-IF]");
105                     aoStatements.add(new IfThenStatement(oCondition, iOrgPosition, oEndIfToken.getLine()));
106                     break;
107 
108                 // INPUT Token: Read the line from terminal for processing
109                 case INPUT:
110                     _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [INPUT] ");
111                     oLineNumber.putLineNumber(getToken(0).getLine(), _iPosition);
112                     _iPosition++;
113                     aoStatements.add(new InputStatement(_iPosition -1, consumeToken(TokenType.WORD).getText()));
114                     break;
115 
116                 // LABEL Token: tbd
117                 case LABEL:
118                     _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [LABEL] ");
119                     oLabelStatement.putLabelStatement(getToken(0).getText(), aoStatements.size());
120                     _iPosition++;
121                     break;
122 
123                 // LINE Token: Describe an empty line, ignore
124                 case LINE:
125                     _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [LINE] ");
126                     _iPosition++;
127                     break;
128 
129                 // PRINT Token: print to the terminal
130                 case PRINT:
131                     _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [PRINT] ");
132                     oLineNumber.putLineNumber(getToken(0).getLine(), _iPosition);
133                     _iPosition++;
134                     aoStatements.add(new PrintStatement(_iPosition -1, expression()));
135                     break;
136 
137                 // REM Token: contains comments to the program, ignore the rest of the line
138                 case REM:
139                     _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [REM] ");
140                     _iPosition++;
141                     break;
142 
143                 // WORD Token: This word is a variable, anything following is variable manipulation
144                 case WORD:
145                     _oLogger.debug("-parse-> found Token: <" + _iPosition + "> [WORD] ");
146                     int iJumpPosition = _iPosition;
147                     oLineNumber.putLineNumber(getToken(0).getLine(), _iPosition);
148 
149                     if (getToken(1).getType() == TokenType.ASSIGN_EQUAL) {
150                         String strName = getToken(0).getText();
151                         _iPosition = _iPosition + 2;
152                         Expression oValue = expression();
153                         aoStatements.add(new AssignStatement(iJumpPosition, strName, oValue));
154                     } else {
155                         throw new SyntaxErrorException("Incorrect Operator: " + getToken(0).getType().toString() +
156                                 " in Line [" + getToken(0).getLine() + "]");
157                     }
158 
159                     break;
160 
161                 // List of all tokens that are used in a different context (e.g. as part of the IFTHEN token)
162                 case ENDIF:
163                     _oLogger.debug("-parse-> ignoring token: <" + _iPosition + "> [" + getToken(0).getType() + "] ");
164                     _iPosition++;
165                     break;
166 
167                 // No Token identified, Syntax Error
168                 default:
169                     throw new SyntaxErrorException("Incorrect Command: " + getToken(0).getLine() + ": ["
170                             + getToken(0).getType() + "] <"
171                             + getToken(0).getLine() + ">");
172             }
173         }
174 
175         for (Statement oStatement: aoStatements) {
176             oLineNumber.putStatementNumber(oStatement.getLineNumber(), aoStatements.indexOf(oStatement));
177         }
178 
179         return aoStatements;
180     }

182     // The following functions each represent one grammatical part of the
183     // language. If this parsed English, these functions would be named like
184     // noun() and verb().

186     /**
187      * Parses a single expression. Recursive descent parsers start with the
188      * lowest-precedent term and moves towards higher priority. For Basic,
189      * binary operators (+, -, etc.) are the lowest.
190      *
191      * @return The parsed expression.
192      * @throws SyntaxErrorException - marks any syntax errors
193      */
194     private Expression expression() throws SyntaxErrorException {
195         _oLogger.debug("-expression-> <" + _iPosition + "> [" + getToken(0).getType().toString() + "]");
196         return operator();
197     }
198 
199     /**
200      * Parses a series of binary operator expressions into a single
201      * expression. In Basic, all operators have the same predecence and
202      * associate left-to-right. That means it will interpret:
203      *    1 + 2 * 3 - 4 / 5
204      * like:
205      *    ((((1 + 2) * 3) - 4) / 5)
206      *
207      * It works by building the expression tree one at a time. So, given
208      * this code: 1 + 2 * 3, this will:
209      *
210      * 1. Parse (1) as an atomic expression.
211      * 2. See the (+) and start a new operator expression.
212      * 3. Parse (2) as an atomic expression.
213      * 4. Build a (1 + 2) expression and replace (1) with it.
214      * 5. See the (*) and start a new operator expression.
215      * 6. Parse (3) as an atomic expression.
216      * 7. Build a ((1 + 2) * 3) expression and replace (1 + 2) with it.
217      * 8. Return the last expression built.
218      *
219      * @return The parsed expression.
220      * @throws SyntaxErrorException - marks any syntax issues
221      */
222     public final Expression operator() throws SyntaxErrorException {
223         Expression oExpression = atomic();
224 
225         // Keep building operator expressions as long as we have operators.
226         Token oToken = getToken(0);
227         _oLogger.debug("-operator-> token: <" + _iPosition + "> [" + oToken.getType().toString() + "] '" + oToken.getText() + "' [" + oToken.getLine() + "]");
228 
229         // loop while the token inspected is either an operator or an assignment
230         while ((oToken.getType() == TokenType.PLUS) ||
231                 (oToken.getType() == TokenType.MINUS) ||
232                 (oToken.getType() == TokenType.MULTIPLY) ||
233                 (oToken.getType() == TokenType.DIVIDE) ||
234                 (oToken.getType() == TokenType.POWER) ||
235                 (oToken.getType() == TokenType.COMPARE_EQUAL) ||
236                 (oToken.getType() == TokenType.ASSIGN_EQUAL)) {
237             _oLogger.debug("-operator-> token: <" + _iPosition + "> [" + oToken.getType().toString() + "] '" + oToken.getText() + "' [" + oToken.getLine() + "]");
238             _iPosition++;
239             Expression oRight = atomic();
240             oExpression = new OperatorExpression(oExpression, oToken.getText(), oRight);
241             oToken = getToken(0);
242         }
243 
244         _oLogger.debug("-operator-> expression: [" + oExpression.content() + "]");
245         return (oExpression);
246     }

248     /**
249      * Parses an "atomic" expression. This is the highest level of
250      * precedence and contains single literal tokens like 123 and "foo", as
251      * well as parenthesized expressions.
252      *
253      * @return The parsed expression.
254      * @throws SyntaxErrorException - mark any syntax issues
255      */
256     public final Expression atomic() throws SyntaxErrorException {
257         Token oToken = getToken(0);
258 
259         _oLogger.debug("-atomic-> search token: <" + _iPosition + "> [" + oToken.getType().toString() + "] ");
260 
261         switch (oToken.getType()) {
262 
263             // If the current token is of type WORD, then we assume it is a variable.
264             case WORD:
265                 oToken = getToken(0);
266                 _oLogger.debug("-atomic-> found token: <" + _iPosition + "> [" + oToken.getType().toString() + "] '" + oToken.getText() + "' [" + oToken.getLine() + "]");
267                 _iPosition++;
268                 return (new VariableExpression(oToken.getText()));
269 
270             // If the current token is of type NUMBER, then return the value as a double value
271             case NUMBER:
272                 oToken = getToken(0);
273                 _oLogger.debug("-atomic-> found token: <" + _iPosition + "> [" + oToken.getType().toString() + "] '" + oToken.getText() + "' [" + oToken.getLine() + "]");
274                 _iPosition++;
275                 return new RealValue(Double.parseDouble(oToken.getText()));
276 
277             // If the current token us of type STRING, then return the value as a string value
278             case STRING:
279                 oToken = getToken(0);
280                 _oLogger.debug("-atomic-> found token: <" + _iPosition + "> [" + oToken.getType().toString() + "] '" + oToken.getText() + "' [" + oToken.getLine() + "]");
281                 _iPosition++;
282                 return new StringValue(oToken.getText());
283 
284             // The contents of a parenthesized expression can be any expression. This lets us "restart" the precedence cascade
285             // so that you can have a lower precedence expression inside the parentheses.
286             case LEFT_PAREN:
287                 oToken = getToken(0);
288                 _oLogger.debug("-atomic-> found token: <" + _iPosition + "> [" + oToken.getType().toString() + "] '" + oToken.getText() + "' [" + oToken.getLine() + "]");
289                 _iPosition++;
290                 Expression expression = expression();
291                 consumeToken(TokenType.RIGHT_PAREN);
292                 return (expression);
293         }
294 
295         // The contents of a parenthesized expression can be any expression. This lets us "restart" the precedence cascade
296         // so that you can have a lower precedence expression inside the parentheses.
297         //if (matchNextToken(TokenType.LEFT_PAREN)) {
298         //    Expression expression = expression();
299         //    consumeToken(TokenType.RIGHT_PAREN);
300         //    return (expression);
301         //}
302 
303         // OK - here we have a text block that we cannot parse, so we throw an syntax exception
304         throw new SyntaxErrorException("Couldn't parse : [" + getToken(0).getType().toString() + "] <" + getToken(0).getLine() + ">");
305     }

307     /**
308      * Consumes the next token if it's a word token with the given name.
309      *
310      * @param  strName  Expected name of the next word token.
311      * @return          True if the token was consumed.
312      */
313     public final boolean matchNextToken(final String strName) {
314 
315         if (getToken(0).getType() != TokenType.WORD) {
316             _oLogger.debug("-matchNextToken-> token compare failed");
317             return (false);
318         }
319 
320         if (!getToken(0).getText().equals(strName)) {
321             _oLogger.debug("-matchNextToken-> token compare failed");
322             return (false);
323         }
324 
325         _iPosition++;
326         _oLogger.debug("-matchNextToken-> token compare success");
327         return (true);
328     }

330     /**
331      * Consumes the next token if it's the given type. If not, throws an
332      * exception. This is for cases where the parser demands a token of a
333      * certain type in a certain position, for example a matching ) after
334      * an opening (.
335      *
336      * @param  oType  Expected type of the next token.
337      * @return       The consumed token.
338      */
339     public final Token consumeToken(final TokenType oType) throws SyntaxErrorException {
340         _oLogger.debug("-consumeToken-> looking for Token: <" + (_iPosition) + "> ["+ oType.toString() + "]: <" + getToken(0).getType() +
341                 ">");
342 
343         if (getToken(0).getType() != oType) {
344             throw new SyntaxErrorException("Expected " + oType + ".");
345         }
346 
347         return (_aoTokens.get(_iPosition++));
348     }
349 
350     /**
351      * Consumes the next token if it's the given type. If not, throws an
352      * exception. This is for cases where the parser demands a token of a
353      * certain type in a certain position, for example a matching ) after
354      * an opening (.
355      *
356      * @param  oType  Expected type of the next token.
357      * @return       The found token.
358      */
359     public final Token findToken(final TokenType oType) throws SyntaxErrorException {
360         _oLogger.debug("-findToken-> looking for Token: <" + (_iPosition) + "> ["+ oType.toString() + "]: <" + getToken(0).getType() +
361                 ">");
362 
363         int iCurrentPosition = _iPosition;
364 
365         while (iCurrentPosition < _aoTokens.size()) {
366             Token oToken = _aoTokens.get(iCurrentPosition);
367 
368             if (oToken.getType() == oType) {
369                 _oLogger.debug("-findToken-> found: <" + (iCurrentPosition) + "> ["+ oToken.getType() + "]");
370                 return (oToken);
371             } else {
372                 _oLogger.debug("-findToken-> failed: <" + (iCurrentPosition) + "> ["+ oToken.getType() + "]");
373                 iCurrentPosition++;
374             }
375         }
376 
377         throw new SyntaxErrorException("Missing statement " + oType + ".");
378     }

380     /**
381      * Gets an unconsumed token, indexing forward. get(0) will be the next
382      * token to be consumed, get(1) the one after that, etc.
383      *
384      * @param  iOffset How far forward in the token stream to look.
385      * @return        The yet-to-be-consumed token.
386      */
387     public final Token getToken(final int iOffset) {
388 
389         //check whether the current position in the tokenized program is larger or equal the token size
390         if (_iPosition + iOffset >= _aoTokens.size()) {
391             // send an end_of_file token back - this is an unexpected EOP
392             // TODO actually this is a syntax error and should throw the syntax error exception
393             return (new Token("", TokenType.EOP, 0));
394         }
395 
396         // get the requested token
397         return (_aoTokens.get(_iPosition + iOffset));
398     }
