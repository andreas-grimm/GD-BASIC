3   import eu.gricom.interpreter.basic.variableTypes.Value;

5   /**
6    * Base interface for an expression. An expression is like a statement
7    * except that it also returns a value when executed. Expressions do not
8    * appear at the top level in Jasic programs, but are used in many
9    * statements. For example, the value printed by a "print" statement is an
10   * expression. Unlike statements, expressions can nest.
11   */
12  
13  public interface Expression {
14      /**
15       * Expression classes implement this to evaluate the expression and
16       * return the value.
17       *
18       * @return The value of the calculated expression.
19       * @throws Exception - Any Exception generated in a lower part of the program.
20       */
21      Value evaluate() throws Exception;
22  
23      /**
24       * Get the content in human readable form.
25       *
26       * @return Content in readable form
27       */
28      String content();
29  }
