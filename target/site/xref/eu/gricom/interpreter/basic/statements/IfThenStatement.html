1   package eu.gricom.interpreter.basic.statements;
2   
3   import eu.gricom.interpreter.basic.memoryManager.ProgramPointer;
4   import eu.gricom.interpreter.basic.variableTypes.BooleanValue;
5   
6   /**
7    * IfThenStatement.java
8    * <p>
9    * Description:
10   * <p>
11   * An if then statement jumps execution to another place in the program, but only if an expression evaluates to
12   * something other than 0.
13   * <p>
14   * (c) = 2004,..,2016 by Andreas Grimm, Den Haag, The Netherlands
15   * <p>
16   * Created in 2003
17   *
18   */
19  public final class IfThenStatement implements Statement {
20  
21      private final Expression _oCondition;
22      private final String _strLabel;
23      private int _iStatementNumber = 0;
24      private int _iEndLocation = 0;
25      private ProgramPointer _oProgramPointer = new ProgramPointer();
26      private LabelStatement _oLabelStatement = new LabelStatement();
27  
28      /**
29       * JASIC constructor.
30       *
31       * @param oCondition - condition to be tested.
32       * @param strLabel - destination for the jump after successful completion of the condition.
33       */
34      public IfThenStatement(final Expression oCondition, final String strLabel) {
35          _oCondition = oCondition;
36          _strLabel = strLabel;
37      }
38  
39      /**
40       * BASIC constructor.
41       *
42       * @param oCondition - condition to be tested.
43       * @param iStatementNumber - the sequence number of this statement in the program
44       * @param iEndLocation - destination for the jump after unsuccessful completion of the condition.
45       */
46      public IfThenStatement(final Expression oCondition, final int iStatementNumber, final int iEndLocation) {
47          _iStatementNumber = iStatementNumber;
48          _oCondition = oCondition;
49          _strLabel = "";
50          _iEndLocation = iEndLocation;
51      }
52  
53      /**
54       * Default constructor.
55       *
56       * @param oCondition - condition to be tested.
57       * @param strLabel - destination for the jump after successful completion of the condition.
58       */
59      public IfThenStatement(final int iLineNumber, final Expression oCondition, final String strLabel) {
60          _iStatementNumber = iLineNumber;
61          _oCondition = oCondition;
62          _strLabel = strLabel;
63          _iEndLocation = 0;
64      }
65  
66      /**
67       * Get Line Number.
68       *
69       * @return iLineNumber - the command line number of the statement
70       */
71      @Override
72      public int getLineNumber() {
73          return (_iStatementNumber);
74      }
75  
76      /**
77       * Execute the If statement.
78       *
79       * @throws Exception - exposes any exception coming from the memory management
80       */
81      public void execute() throws Exception {
82          if (_oLabelStatement.containsLabelKey(_strLabel)) {
83              BooleanValue bValue = (BooleanValue) _oCondition.evaluate();
84              if (bValue.isTrue()) {
85                  _oProgramPointer.setCurrentStatement(_oLabelStatement.getLabelStatement(_strLabel));
86              }
87          }
88      }
89  
90      /**
91       * This method is used in testing and debugging. It returns the set values when the constructor has been called.
92       *
93       * @return - readable string with the name and the value of the assignment
94       */
95      @Override
96      public String content() {
97  
98          return ("IF (" + _oCondition.content() + ") THEN " + _strLabel);
99      }
100 }
