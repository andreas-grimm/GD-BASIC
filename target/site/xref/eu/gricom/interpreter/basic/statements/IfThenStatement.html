7   /**
8    * IfThenStatement.java
9    * <p>
10   * Description:
11   * <p>
12   * An if then statement jumps execution to another place in the program, but only if an expression evaluates to
13   * something other than 0.
14   * <p>
15   * (c) = 2004,..,2021 by Andreas Grimm, Den Haag, The Netherlands
16   * <p>
17   * Created in 2021
18   *
19   */
20  public final class IfThenStatement implements Statement {
21      private final Expression _oCondition;
22      private final String _strLabel;
23      private int _iStatementNumber = 0;
24      private final ProgramPointer _oProgramPointer = new ProgramPointer();
25      private final LabelStatement _oLabelStatement = new LabelStatement();
26      private final LineNumberStatement _oLineNumberObject = new LineNumberStatement();
27      private final int _iEndIfLine;
28  
29      /**
30       * JASIC constructor.
31       *
32       * @param oCondition - condition to be tested.
33       * @param strLabel - destination for the jump after successful completion of the condition.
34       */
35      public IfThenStatement(final Expression oCondition, final String strLabel) {
36          _oCondition = oCondition;
37          _strLabel = strLabel;
38          _iEndIfLine = 0;
39      }
40  
41      /**
42       * BASIC constructor.
43       *
44       * @param oCondition - condition to be tested.
45       * @param iStatementNumber - the sequence number of this statement in the program
46       * @param iEndIfLine - destination for the jump after unsuccessful completion of the condition.
47       */
48      public IfThenStatement(final Expression oCondition, final int iStatementNumber, final int iEndIfLine) {
49          _iStatementNumber = iStatementNumber;
50          _oCondition = oCondition;
51          _strLabel = "";
52          _iEndIfLine = iEndIfLine;
53      }
54  
55      /**
56       * Get Line Number.
57       *
58       * @return iLineNumber - the command line number of the statement
59       */
60      @Override
61      public int getLineNumber() {
62          return (_iStatementNumber);
63      }
64  
65      /**
66       * Execute the If statement.
67       *
68       * @throws Exception - exposes any exception coming from the memory management
69       */
70      public void execute() throws Exception {
71          // This part of the method is executed if the BASIC interpreter uses labels (e.g. we are using JASIC)
72          if (_oLabelStatement.containsLabelKey(_strLabel)) {
73              BooleanValue bValue = (BooleanValue) _oCondition.evaluate();
74              if (bValue.isTrue()) {
75                  _oProgramPointer.setCurrentStatement(_oLabelStatement.getLabelStatement(_strLabel));
76              }
77  
78              return;
79          }
80  
81          // here we are using line numbers to jump to the destination. This is only done for BASIC programs.
82          BooleanValue bValue = (BooleanValue) _oCondition.evaluate();
83  
84          // different to the code above: when the result of the condition is false, then ignore the next block and
85          // jump to the END-IF statement.
86          if (!bValue.isTrue()) {
87              try {
88  
89                  if (_iEndIfLine != 0) {
90                      int iStatementNo =
91                              _oLineNumberObject.getStatement(_oLineNumberObject.getNextStatement(_iEndIfLine));
92  
93                      if (iStatementNo != 0) {
94                          _oProgramPointer.setCurrentStatement(iStatementNo);
95                          return;
96                      }
97  
98                      return;
99                  }
100 
101                 throw (new SyntaxErrorException("IF [unknown]: Target: [" + _iEndIfLine + "]"));
102             } catch (NumberFormatException eNumberException) {
103                 throw (new SyntaxErrorException("IF [incorrect format]: Target: " + _strLabel));
104             }
105         }
106 
107     }
108 
109     /**
110      * This method is used in testing and debugging. It returns the set values when the constructor has been called.
111      *
112      * @return - readable string with the name and the value of the assignment
113      */
114     @Override
115     public String content() {
116 
117         return ("IF (" + _oCondition.content() + ") THEN " + _strLabel);
118     }
119 }
