3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * JasicLexer.java
8    * <p>
9    * Description:
10   * <p>
11   * The Tokenizer class converts the source code into a list of tokens. Question here is whether we should convert
12   * the Tokenizer class not into a Token factory class. The read a source code file and move the content into a single
13   * string. This might cause problems in case of really big programs, so this needs to be addressed later.
14   * <p>
15   * (c) = 2020,.., by Andreas Grimm, Den Haag, The Netherlands
16   */
17  public final class JasicLexer implements Tokenizer {
18      private int _iLineNumber = 1;
19  
20      /**
21       * Default constructor, private as due to a utility class.
22       */
23      public JasicLexer() {
24      }
25  
26      @Override
27      public List<Token> tokenize(final String strSource) {
28          List<Token> aoTokens = new ArrayList<>();
29  
30          String strToken = "";
31          TokenizeState oState = TokenizeState.DEFAULT; //The token has not been identified - so the token type is default
32  
33          // Many tokens are a single character, like operators and ().
34          String charTokens = "\n=+-*/<>()";
35          TokenType[] eTokenTypes = {
36                  TokenType.LINE,
37                  TokenType.EQUALS,
38                  TokenType.OPERATOR,
39                  TokenType.OPERATOR,
40                  TokenType.OPERATOR,
41                  TokenType.OPERATOR,
42                  TokenType.OPERATOR,
43                  TokenType.OPERATOR,
44                  TokenType.LEFT_PAREN,
45                  TokenType.RIGHT_PAREN};
46  
47          // Scan through the code one character at a time, building up the list of tokens.
48          for (int i = 0; i < strSource.length(); i++) {
49              char c = strSource.charAt(i);
50              switch (oState) {
51                  case DEFAULT:
52                      if (charTokens.indexOf(c) != -1) {
53                          aoTokens.add(new Token(Character.toString(c), eTokenTypes[charTokens.indexOf(c)], _iLineNumber));
54                          if (eTokenTypes[charTokens.indexOf(c)] == TokenType.LINE) {
55                              // if the found character is a 'new line' then increase the line number.
56                              _iLineNumber++;
57                          }
58                      } else if (Character.isLetter(c)) {
59                          strToken += String.valueOf(c);
60                          oState = TokenizeState.WORD;
61                      } else if (Character.isDigit(c)) {
62                          strToken += String.valueOf(c);
63                          oState = TokenizeState.NUMBER;
64                      } else if (c == '"') {
65                          oState = TokenizeState.STRING;
66                      } else if (c == '\'') {
67                          oState = TokenizeState.COMMENT;
68                      }
69                      break;
70  
71                  case WORD:
72                      if (Character.isLetterOrDigit(c)) {
73                          strToken += String.valueOf(c);
74                      } else if (c == ':') {
75                          aoTokens.add(new Token(strToken, TokenType.LABEL, _iLineNumber));
76                          strToken = "";
77                          oState = TokenizeState.DEFAULT;
78                      } else {
79                          aoTokens.add(new Token(strToken, TokenType.WORD, _iLineNumber));
80                          strToken = "";
81                          oState = TokenizeState.DEFAULT;
82                          i--; // Reprocess this character in the default state.
83                      }
84                      break;
85  
86                  case NUMBER:
87                      // HACK: Negative numbers and floating points aren't supported.
88                      // To get a negative number, just do 0 - <your number>.
89                      // To get a floating point, divide.
90                      if (Character.isDigit(c)) {
91                          strToken += String.valueOf(c);
92                      } else {
93                          aoTokens.add(new Token(strToken, TokenType.NUMBER, _iLineNumber));
94                          strToken = "";
95                          oState = TokenizeState.DEFAULT;
96                          i--; // Reprocess this character in the default state.
97                      }
98                      break;
99  
100                 case STRING:
101                     if (c == '"') {
102                         aoTokens.add(new Token(strToken, TokenType.STRING, _iLineNumber));
103                         strToken = "";
104                         oState = TokenizeState.DEFAULT;
105                     } else {
106                         strToken += String.valueOf(c);
107                     }
108                     break;
109 
110                 case COMMENT:
111                     if (c == '\n') {
112                         oState = TokenizeState.DEFAULT;
113                     }
114                     break;
115 
116                 default:
117                     break;
118             }
119         }
120         // TODO - This needs to be fixed...
121         // HACK: Silently ignore any in-progress token when we run out of
122         // characters. This means that, for example, if a script has a string
123         // that's missing the closing ", it will just ditch it.
124         return (aoTokens);
125     }
126 }
