30      ERL,
31      ERR,
32      EXP,
33      FOR,
34      FRE,
35      GOSUB,
36      GOTO,
37      IF,
38      INSTR,

40      INPUT,

94      INTEGER, // Token for any integer numbers
95      BOOLEAN, // Token for any boolean variable
96      OPERATOR, // super-token to cover all operations in JASIC
97      LABEL, // JASIC GOTO label, not supported for the Dartmouth BASIC dialects
98      LINE, // empty line - white space. Kept to keep the line numbering consistent
99      ASSIGN, // duplication of the ASSIGN_EQUAL and COMPARE_EQUAL token - to be removed
100     EQUALS
