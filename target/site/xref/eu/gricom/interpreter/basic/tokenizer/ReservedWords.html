55      public static int getIndex (final String strSearchRole) {
56          int iReturn = -1;
57          int iIndex = 0;
58  
59          for (String strReserveWord: astrReservedWords) {
60              if (strSearchRole.toUpperCase(Locale.ROOT).matches(strReserveWord)) {
61                  iReturn = iIndex;
62                  break;
63              }
64  
65              iIndex++;
66          }
67  
68          return (iReturn);
69      }
70  
71      public static TokenType getTokenType (int iIndex) {
72          return (aeTokenTypes[iIndex]);
73      }
