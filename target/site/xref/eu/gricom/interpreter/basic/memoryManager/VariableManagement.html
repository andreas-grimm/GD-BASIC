10  /**
11   * VariableManagement.java
12   * <p>
13   * Description:
14   * This structure holds all variables used in the Basic program.
15   * <p>
16   * The Program Pointer contains the current position of the program.

18   * (c) = 2020,.., by Andreas Grimm, Den Haag, The Netherlands
19   */
20  public class VariableManagement {
21      private static Map<String, Value> _aoVariables = new HashMap<>();
22  
23      /**
24       * Default Constructor.
25       */
26      public VariableManagement() {
27      }
28  // section managing internal variables...
29      /**
30       * Put a key - value pair into the variable map structure.
31       *
32       * @param strName - key part of the pair
33       * @param dValue - value part of the pair, here as an double
34       */
35      public final void putMap(final String strName, final double dValue) {
36          RealValue oValue = new RealValue(dValue);
37          _aoVariables.put(strName, oValue);
38      }
39  
40      /**
41       * Put a key - value pair into the variable map structure.
42       *
43       * @param strName - key part of the pair
44       * @param strValue - value part of the pair, here as a string
45       */
46      public final void putMap(final String strName, final String strValue) {
47          StringValue oValue = new StringValue(strValue);
48          _aoVariables.put(strName, oValue);
49      }
50  
51      /**
52       * Get variable defined by a given key value.
53       *
54       * @param strKey - Key used for retrieval
55       * @return Value object to be returned
56       */
57      public final Value getMap(final String strKey) {
58  
59          return (_aoVariables.get(strKey));
60      }
61  
62      /**
63       * Verifies that the variables structure contains a given key.
64       *
65       * @param strKey - Key to be verified
66       * @return true, if key is in the data structure
67       */
68      public final boolean mapContainsKey(final String strKey) {
69  
70          return (_aoVariables.containsKey(strKey));
71      }
72  }
